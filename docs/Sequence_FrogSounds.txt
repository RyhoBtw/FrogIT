@startuml

actor User
participant "FrogSettingsUI" as UI
participant "Soundscape" as SS
participant "Sound" as Sound
participant "Resource" as Resource

User-> UI : Adjust Sound Settings
UI -> SS : playAll()

loop for each active sound
    SS -> Sound : play()
    activate Sound
    Sound -> Resource : load(ResourceType.SOUND)
    Resource --> Sound : return loaded sound data
    Sound -> Sound : start audio stream (volume, frequency)
    deactivate Sound
end

SS --> UI : playback started confirmation
UI -> User: Display "Playing soundscape"

@enduml
